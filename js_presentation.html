ES5 inheritance


        
            function Point(x, y){
                this.x = x;
                this.y = y;
            }
            Point.prototype.toString = function() {
                return '(' + this.x + ', ' + this.y + ')';
            }
            function ColorPoint(x, y, color) {
                Point.call(this, x, y);
                this.color = color;
            }
            ColorPoint.prototype = Object.create(Point.prototype);
            ColorPoint.prototype.constructor = ColorPoint;
            ColorPoint.prototype.toString = function() {
                return Point.prototype.toString.call(this, this.x, this.y) + this.color;
            }
            var cp = new  ColorPoint(10, 10, 'red');

            class Animal {

                constructor(theName: string) { this.name = theName; }
                protected move(distanceInMeters: number = 0) {
                    console.log(`${this.name} moved ${distanceInMeters}m.`);
                }
            }
             
            class Horse extends Animal {
                private age: number;
                constructor(name: string) { super(name); }
                protected move(distanceInMeters = 45) {
                    console.log("Galloping...");
                    super.move(distanceInMeters);
                }
            }
 
            const tom: Animal = new Horse("Horse");
